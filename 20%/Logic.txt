    function isPalindrome(str) {
    let left = 0;
    let right = str.length - 1;

    while (left < right) {
        if (str[left] !== str[right]) {
            return false;
        }
        left++;
        right--;
    }

    return true;
    }

  if (n <= 1) return false;

14. Create a function which will accepts two arrays arr1 and arr2. The function should return true if every value in arr1 has
its corresponding value squared in array2. The frequency of values must be same.(Effecient)

üÖ∞Ô∏è solution 1 :-

    function same(arr1, arr2) {
    if (arr1.length !== arr2.length) return false;

    const freq1 = {};
    const freq2 = {};

    // Count frequencies
    for (let val of arr1) {
        freq1[val] = (freq1[val] || 0) + 1;
    }

    for (let val of arr2) {
        freq2[val] = (freq2[val] || 0) + 1;
    }

    // Compare squares
    for (let key in freq1) {
        const squared = key * key;

        if (!(squared in freq2)) return false;               // checks key
        if (freq2[squared] !== freq1[key]) return false;     // checks value  // checks frequency
    }

    return true;
    }

    console.log(same([1,2,2], [1,4,4])); // true
    console.log(same([1,2,3], [1,9]));   // false

    for (let i = 0; i < arr1.length; i++) {
        const index = arr2.indexOf(arr1[i] * arr1[i]);

    for (let char of str1) {

    for (let i = 0; i < arr.length; i++) {
        for (let j = 0; j < arr.length - i - 1; j++) {
        if (arr[j] > arr[j + 1]) {
