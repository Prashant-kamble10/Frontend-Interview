🤔 sum of all odd numbers in an array

🅰️  let arr = [5, 3, 2, 0, 1]
        let len = arr.length - 1

        function sample(a) {
            if (a < 0) return 0; // base case to stop recursion

            if (arr[a] % 2 != 0) {
                return arr[a] + sample(a - 1)
            } else {
            return sample(a - 1) // skip even numbers
            }
        }

        console.log(sample(len)) // Output: 9 (5 + 3 + 1)
